        def restart_program():

            a=int(input("Enter A Number: "))
            s=0
            for i in range (1,a//2+1): #Finding factors of "a" from 1 to half of "a" + 1 for faster code
              if a%i == 0:
                s = s+i # "s" initially 0 but, for every factor of "a", it gets added to "s" and as "s" value gets over written so for every factor it is getting added with the previous factor value written in "s"
            if s == a: # Final Checking if the sum of factors is equal to th number itself#
                 print("Perfect")
            else: # If not same then it is not perfect
                 print("Not Perfect")
             
        while True: #  Looping the program
