        def restart_program():
  
        a = input("Enter The Sentence: ").lower()
        s = 0  
        for l in a: # l is the variable which stores the value for each vowel in string "a"
            if l in ["a","e","i","o","u"]: #Checks if l is any of the vowels from a,e,i,o,u in string "a"
                s += 1 #If vowel is there it adds 1 to value of s which was previously set to 0
        if s == 0:
            print("No vowels in the sentence")
        else:
            print("Vowels in sentence is/are: ", s)
     
        while True:
     
            z = input("Do you want to start the program? [Y/N]: ")
            if z.lower() == "y":
                print("Starting.....\n 1 \n 2 \n 3")
                restart_program()
            elif z.lower() == "n":
                print("Exiting......")
                print("Exited.....",sep=("%^--%$"),end="\n Thank you for using this program")
                exit()
            else:
                print("Invalid Input /n Enter Y or N ")
                continue
